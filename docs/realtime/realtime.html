 74    async def connect(self) -> None:
 75        """Establishes a WebSocket connection with retry logic."""
 76        if self.is_connected:
 77            return
 78        retries = 0
 79        last_exception = None
 80        backoff = self.initial_backoff
 81
 82        # Prepare the WebSocket URL with the auth token
 83        websocket_url = http_to_websocket_url(
 84            self.url,
 85            params={"apikey": self.access_token} if self.access_token else {},
 86        )
 87
 88        while retries < self.max_retries:
 89            try:
 90                ws = await websockets.connect(websocket_url)
 91                # Only modify shared state after successful connection
 92                self._ws = ws
 93                self.is_connected = True
 94                logger.info(f"WebSocket connection established to {websocket_url}.")
 95                self._receive_task = asyncio.create_task(self._receive_loop())
 96                self._heartbeat_task = asyncio.create_task(self._heartbeat_loop())
 97                self._reconnect_timer.reset()
 98                return
 99            except Exception as e:
100                retries += 1
101                last_exception = e
102                logger.warning(f"Connection attempt {retries} failed: {e}")
103                await asyncio.sleep(backoff)
104                backoff *= 2
105        logger.error(f"Failed to establish connection after {retries} attempts.")
106        raise ConnectionFailedError(retries, last_exception)

180        """Handles a received message from the server."""
181        msg = Message(**message)

199        """Sends heartbeat messages at regular intervals."""
200        while self.is_connected:

212                break  # Exit the loop as reconnection will be handled

214            await asyncio.sleep(self.hb_interval)

216    async def _handle_connection_error(self, error: Exception) -> None:
217        """Handles connection errors and initiates reconnection if enabled."""
218        logger.exception(f"Connection error occurred: {error}")
219        if self.is_connected:
220            self.is_connected = False
221            # Notify channels of disconnection
222            for channel in self._channels.values():
223                await channel._on_disconnect()
224            if self.auto_reconnect and not self.is_reconnecting:
225                self.is_reconnecting = True
226                self._reconnect_timer.reset()
227                self._reconnect_timer.schedule()
228            else:
229                await self.disconnect()

231    async def _attempt_reconnect(self) -> None:
232        """Attempts to reconnect to the server."""
233        logger.info("Attempting to reconnect to the WebSocket server.")
234        if self.is_reconnecting:
235            return
236        self.is_reconnecting = True
237        try:
238            await self.disconnect()
239            await self.connect()
240            # Rejoin channels
241            for channel in self._channels.values():
242                if channel.auto_rejoin:
243                    await channel.subscribe()
244                else:
245                    logger.debug(f"Auto rejoin is disabled for channel: {channel.topic}")
246        except Exception as e:
247            logger.exception(f"Reconnection attempt failed: {e}")
248        finally:
249            self.is_reconnecting = False

166    async def _sync_state(
167        self, current_state: PresenceStateType, new_state: Dict[str, Any]
168    ) -> PresenceStateType:
169        """Synchronizes the current presence state with the new state received from the server."""
170        # Transform the new presence data into Presence instances
171        transformed_new_state = self._transform_state(new_state)
172
173        # Will hold the updated state after synchronization
174        state: PresenceStateType = {}
175
176        # Diffs to hold joins and leaves
177        joins: Dict[str, List[Presence]] = {}
178        leaves: Dict[str, List[Presence]] = {}
179
180        # Get sets of all keys (presence keys) from current and new state
181        current_keys = set(current_state.keys())
182        new_keys = set(transformed_new_state.keys())
183
184        # Calculate leaves (keys present in current state but not in new state)
185        for key in current_keys - new_keys:
186            leaves[key] = current_state[key]
187
188        # Calculate joins (keys present in new state but not in current state)
189        for key in new_keys - current_keys:
190            joins[key] = transformed_new_state[key]
191
192        # For keys present in both current and new state, compare presence refs
193        for key in current_keys & new_keys:
194            current_presences = current_state[key]
195            new_presences = transformed_new_state[key]
196
197            # Build sets of phx_refs for quick lookup
198            current_refs = {presence.phx_ref for presence in current_presences}
199            new_refs = {presence.phx_ref for presence in new_presences}
200
201            # Presences that have joined
202            joined_presences = [
203                presence
204                for presence in new_presences
205                if presence.phx_ref not in current_refs
206            ]
207            if joined_presences:
208                joins[key] = joined_presences
209
210            # Presences that have left
211            left_presences = [
212                presence
213                for presence in current_presences
214                if presence.phx_ref not in new_refs
215            ]
216            if left_presences:
217                leaves[key] = left_presences
218
219        # Update the state with new presences
220        # For all keys, the presences are replaced with the ones from new state
221        state.update(transformed_new_state)
222
223        # Build the diff dictionary
224        diff = {"joins": joins, "leaves": leaves}
225
226        # Apply the diff (this will trigger callbacks)
227        await self._apply_diff(state, diff)
228
229        return state
230
231    async def _sync_diff(
232        self, current_state: PresenceStateType, diff: Dict[str, Any]
233    ) -> PresenceStateType:
234        """Applies diffs to the presence state."""
235        state = current_state.copy()
236        joins_raw: Dict[str, Any] = diff.get("joins", {})
237        leaves_raw: Dict[str, Any] = diff.get("leaves", {})
238
239        # Transform joins and leaves into Presence instances
240        joins = self._transform_state(joins_raw)
241        leaves = self._transform_state(leaves_raw)
242
243        # Build the diff with transformed joins and leaves
244        diff_transformed = {"joins": joins, "leaves": leaves}
245
246        # Apply the diff
247        await self._apply_diff(state, diff_transformed)
248
249        return state
250
251    async def _apply_diff(self, state: PresenceStateType, diff: Dict[str, Any]) -> None:
252        """Applies the transformed diff to the state and triggers callbacks."""
253        joins: Dict[str, List[Presence]] = diff.get("joins", {})
254        leaves: Dict[str, List[Presence]] = diff.get("leaves", {})
255
256        # Handle joins
257        for key, new_presences in joins.items():
258            current_presences = state.get(key, [])
259            state[key] = current_presences + new_presences
260
261            # Trigger on_join callbacks
262            for callback in self._callbacks["join"]:
263                try:
264                    await callback(key, current_presences.copy(), new_presences)
265                except Exception as e:
266                    logger.exception(f"Exception in presence join callback: {e}")
267
268        # Handle leaves
269        for key, left_presences in leaves.items():
270            current_presences = state.get(key, [])
271            left_refs = {p.phx_ref for p in left_presences}
272
273            # Remove left presences
274            remaining_presences = [
275                p for p in current_presences if p.phx_ref not in left_refs
276            ]
277            if remaining_presences:
278                state[key] = remaining_presences
279            else:
280                state.pop(key, None)
281
282            # Trigger on_leave callbacks
283            for callback in self._callbacks["leave"]:
284                try:
285                    await callback(key, current_presences.copy(), left_presences)
286                except Exception as e:
287                    logger.exception(f"Exception in presence leave callback: {e}")
288
289        # Update the internal state
290        self.state = state
291
292    def in_pending_sync_state(self) -> bool:
293        """Checks if the presence is in a pending sync state."""
294        return self.join_ref is None or self.join_ref != self.channel.join_ref

296    @staticmethod
297    def _transform_state(state: Dict[str, Any]) -> Dict[str, List[Presence]]:
298        """Transforms the raw presence state into the expected format."""
299        new_state: Dict[str, List[Presence]] = {}
300        for key, presences in state.items():
301            metas = presences.get("metas", [])
302            new_state[key] = []
303            for meta in metas:
304                phx_ref = meta.get("phx_ref")
305                phx_ref_prev = meta.get("phx_ref_prev")
306                # Extract metadata excluding 'phx_ref' and 'phx_ref_prev'
307                metadata = {
308                    k: v
309                    for k, v in meta.items()
310                    if k not in ["phx_ref", "phx_ref_prev"]
311                }
312                presence = Presence(
313                    phx_ref=phx_ref, phx_ref_prev=phx_ref_prev, metadata=metadata
314                )
315                new_state[key].append(presence)
316        return new_state
